#include #include // Include the cstdlib library for srand() and rand() using namespace std; int main() { int month, day; cout << "Welcome to your friendly weather program.\n" << "Enter today's date as two integers for the month and the day:\n"; cin >> month; cin >> day; srand(month * day); // Seed the random number generator int prediction; char ans; cout << "Weather for today:\n"; do { prediction = rand() % 3; // Generate a random number between 0 and 2 switch (prediction) { case 0: cout << "The day will be sunny!!\n"; break; case 1: cout << "The day will be cloudy.\n"; break; case 2: cout << "The day will be stormy!.\n"; break; default: cout << "Weather program is not functioning properly.\n"; } cout << "Want the weather for the next day?(y/n): "; cin >> ans; } while (ans == 'y' || ans == 'Y'); cout << "That's it from your 24 hour weather program.\n"; return 0; }
